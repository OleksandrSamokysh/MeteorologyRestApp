CREATE TABLE Person(
id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, name varchar(100) NOT NULL, age int,
email varchar(100),
created_at timestamp,
                       updated_at timestamp,
                       updatet_who varchar(100)
);

CREATE TABLE Sensor(
                       name varchar(100) PRIMARY KEY UNIQUE NOT NULL,
                       registration_at timestamp
);



CREATE TABLE Data(
                       id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
                       temp_value double PRECISION NOT NULL,
                       raining_status boolean NOT NULL,
                       sensor_name varchar(100) not null references sensor(name),
                       created_at timestamp
);
